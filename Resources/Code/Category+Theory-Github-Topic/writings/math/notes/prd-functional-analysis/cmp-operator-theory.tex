	\startsection [title={Intuition}]
		On a separable Hilbert space, we have
		\startitemize [4, joinedup]
			
			\item  \m{T ‚àà ‚Ñ¨^‚àû ‚ü∫ Œª ‚àà ‚Ñì^‚àû} (bounded)

		\stopitemize

	\startremark

	\stoptheorem

		\startitemize [4]
			
			\item  \m{‚Ñ¨_{00} ‚äÇ ‚Ñ¨^1}  \qquad
				Trivial

			\item  \m{‚Ñ¨^1 ‚äÇ ‚Ñ¨^2}  \qquad

				
			\item  \m{‚Ñ¨^2 ‚äÇ ‚Ñ¨_0}  \qquad

				
			\item  \m{‚Ñ¨_0 ‚äÇ ‚Ñ¨^‚àû}  \qquad
				(\cite[authoryear][BMC2009], Proposition 4.6) If \m{T} is unbounded, we can find a sequence of unit vectors \m{(e_n)} such that \m{\norm{T e_n} ‚Üó ‚àû}. So \m{T e_n} cannot have a convergent subsequence, for if \m{T e_n ‚Üí x}, then \m{\norm{T e_n} ‚Üí \norm{x}}.
				
		\stopitemize
	\stopproof

	\startproposition
		For \m{T ‚àà ‚Ñ¨^‚àû}, \m{\norm{T}_‚àû = \sup\bcrl{\abs{\inn{Tx, y}}} : \norm{x} = 1, \norm{y} = 1}.
	\stopproposition
	\startproof
		\startitemize

			\sym{(‚â§)}  \qquad
				Since \m{\norm{Tx}  =  \frac{\norm{Tx}^2}{\norm{Tx}}  =  \frac{\inn{Tx, Tx}}{\norm{Tx}}  =  \inn{Tx, \frac{Tx}{\norm{Tx}}}}, we have
				\startformula
					\norm{T}_‚àû  =  \sup\bcrl{\norm{Tx} : \norm{x} = 1}  ‚â§  \sup\bcrl{\abs{\inn{Tx, y}} : \norm{x} = 1, \norm{y} = 1} .
				\stopformula
			
			\sym{(‚â•)}  \qquad
				Since \m{\inn{Tx, y}  ‚â§  \norm{Tx} \norm{y}  ‚â§  \norm{T}_‚àû \norm{x} \norm{y}}, we have
				\startformula
					\sup \bcrl{\abs{\inn{Tx, y}} : \norm{x} = 1, \norm{y} = 1}  ‚â§  \norm{T}_‚àû .
				\stopformula
		\stopitemize
	\stopproof

	\startsection [title={Projection operators}]

		\startproposition
			\m{\norm{P}_‚àû ‚â§ 1}.
		\stopproposition
		\startproof
