\startchapter [title={Interpretation of exponential processes}]

	In classical Itô theory, there are three equivalent interpretations of exponential processes.
	\startitemize [i, joinedup]
		\item  renormalization
		\item  martingale
		\item  SDEs
	\stopitemize

	This is not so in the two-sided stochastic integral theory.

\stopchapter

\startchapter [title={Some examples of integrals}]

	\startremark
		For \m{p ∈ ℕ}, we have \m{∫_0^t W_T^p \d W_t = W_T^p W_t - r t W_T^{p-1}}.
	\stopremark
	\startproof
		\startitemize [i]
			
			\item  \bold{From the definition}.
				In the following, \m{\lim} denotes the limit in \m{L^2(Ω)}. We use the quadratic variation of Wiener process, that is, \m{(Δ_i W)^2 → Δ_i t} and \m{(Δ_i W)^{>2} → 0} in \m{L^2}.
				\startformula  \startalign
					\NC  I_t = ∫_0^t W_T^p \d W_u  =  \NC  ∫_0^t \brnd{W_u + (W_T - W_u)}^p \d W_u  \NR
					\NC  =  \NC  \Ltwolim_{\norm{Δ_n} → 0} ∑_{i = 0}^{n-1} \brnd{W_{t_i} + \brnd{W_T - W_{t_{i+1}}}}^p (Δ_i W)  \NR
					\NC  =  \NC  \Ltwolim_{\norm{Δ_n} → 0} ∑_{i = 0}^{n-1} \brnd{W_T - Δ_i W}^p \brnd{Δ_i W}  \NR
					\NC  =  \NC  \Ltwolim_{\norm{Δ_n} → 0} ∑_{i = 0}^{n-1} ∑_{k = 0}^p \binom{p}{k} (-1)^k W_T^{p-k} \brnd{Δ_i W}^{k+1}  \NR
					\NC  =  \NC  ∑_{k = 0}^p \bsqr{ \binom{p}{k} (-1)^k ⋅ \Ltwolim_{\norm{Δ_n} → 0} ∑_{i = 0}^{n-1} W_T^{p-k} \brnd{Δ_i W}^{k+1} }  \NR
					\NC  =  \NC  W_T^p ⋅ \Ltwolim_{\norm{Δ_n} → 0} ∑_{i = 0}^{n-1} \brnd{Δ_i W} - W_T^{p-1} ⋅ \Ltwolim_{\norm{Δ_n} → 0} ∑_{i = 0}^{n-1} \brnd{Δ_i W}^2  \NR
					\NC  =  \NC  W_T^p W_t - p t W_T^{p-1} .  \NR
				\stopalign  \stopformula

			\item  \bold{From the two-sided Itô formula}.
				Since we have to find \m{∫_0^t W_T^p \d W_t}, we guess the solution to be \m{W_t W_T^p}. Decomposing this into adapted and instantly independent parts, we get \m{W_t W_T^p = W_t \brnd{W_t + (W_T - W_t)}^p}. Let \m{X_t = W_t, Y^t = W_T - W_t}. Define \m{ϕ(x, y) = x(x + y)^p}. We have the following partial derivatives.
				\startformula  \startalign[n=4, m=2]
					\NC     ϕ_x =  \NC  (x + y)^p + p x (x + y)^{p-1}                                        \NC     ϕ_y =  \NC  p x (x + y)^{p-1}  \NR
					\NC  ϕ_{xx} =  \NC  p (x + y)^{p-1} + p (x + y)^{p-1} + p (p-1) x (x + y)^{p-2}  \qquad  \NC  ϕ_{yy} =  \NC  p (p-1) x (x + y)^{p-2}  \NR
				\stopalign  \stopformula
				Using the two-sided Itô formula on \m{ϕ(X_t, Y^t)} and noting that \m{\d Y^t = - \d X_t}, we get
				\startformula
					\d (W_t W_T^p) = \d ϕ(X_t, Y^t) = (X_t + Y^t)^p \d W_t + \frac12 ⋅ 2 p (X_t + Y^t)^{p-1} \d t = W_T^p \d W_t + p W_T^{p-1} \d t ,
				\stopformula
				which implies
				\startformula
					∫_0^t W_T^p \d W_s  =  W_t W_T^p - p t W_T^{p-1} .
				\stopformula
		\stopitemize
	\stopproof

	\startremark
		For \m{p ∈ ℕ}, we have \m{∫_0^t W_T^p \d W_t} is a near-martingale.
	\stopremark
	\startproof
		All we need to do is to show that \m{Z_t = 𝔼(I_t ∣ ℱ_t)} is a martingale, where \m{I_t = W_t W_T^p - p t W_T^{p-1}}. For convenience, we write \m{Y_t = W_T - W_t} (note the ad hoc change in notation). Now,
		\startformula
			I_t  =  W_t (W_t + Y_t)^p - p t (W_t + Y_t)^{p-1}  =  W_t ∑_{k = 0}^p \binom{p}{k} W_t^{p - k} Y_t^k - p t ∑_{k = 0}^{p - 1} \binom{p - 1}{k} W_t^{p - 1 - k} Y_t^k .
		\stopformula
		Now, note that \m{W_t} is \m{ℱ_t}-measurable and \m{Y_t} is independent of \m{ℱ_t}, so
		\startformula
			𝔼(Y_t^k ∣ ℱ_t) = 𝔼(Y_t^k) = (k - 1)!! (T - t)^k 𝟙_{k \text{ even}}
		\stopformula
		Now,
		\startformula  \startalign
			\NC  Z_t =  \NC  𝔼(I_t ∣ ℱ_t)  \NR
			\NC      =  \NC  W_t 𝔼(Y_t^p) + (W_t^2 - pt) ∑_{k = 0}^{p - 1} \brnd{\binom{p}{k} + \binom{p - 1}{k}} W_t^{p - k - 1} 𝔼(Y_t^k)  \NR
			\NC      =  \NC  W_t^2 ∑_{k = 0}^{\floor{\frac{p}{2}}} \binom{p}{2k} (2k - 1)!! (T - t)^{2k} W_t^{p - 2k - 1} - p t ∑_{k = 0}^{\floor{\frac{p - 1}{2}}} \binom{p - 1}{2k} (2k - 1)!! (T - t)^{2k} W_t^{p - 2k - 1}  \NR
		\stopalign  \stopformula
